```cpp
#include<iostream>
#include "stack.h"
using namespace std;

int main(int argc, char *argv[]){
  Stack mystack;
  string s;
  cout << "please enter some strings: ";
  while (cin >> s && !mystack.full()) mystack.push(s);
  if (mystack.empty()) 
    cerr << "The stack is empty." << endl;
  else
    cout << "There are " << mystack.size() << " strings in the stack: ";
  while (mystack.peek(s)){
    cout << s << " ";
    mystack.pop(s);
  }
  return 0;

}
```
